22//! ## Storage
23//!
24//! My friend needs to store about 1,5TB of files, mostly photos.
25//! He has 2 laptops with little storage on the internal SSD. We could never use the internal storage to store all of his files. We can use the internal SSD just for "temporary working copies" of some of the files. Eventually we must store the files to a persistent storage.
26//!
27//! 2TB should be enough to store all his files.  
28//! He bought a 2TB super tiny and extremely fast Samsung T7 Shield external SSD. This will be the main persistent storage of his "original files".  
29//! He has also a 2TB old external HDD as the backup drive.  
30//!
31//! ## Original location
32//!
33//! To avoid any confusion, files and folder must have one and only one "original location".  If anything happens, we know where to find our "original files". The fast external SSD is great for that. This will be the persistent storage of all his "original files".
34//!
35//! For some projects it is best to work directly with the "original files". For example when renaming photos and folders.
36//!
37//! ## Temporary working copy
38//!
39//! When working on a project with many small files, first copy the "original files" from the external SSD into the internal "temporary working copy" folder.  
40//!
41//! Work for some time on that files.  
42//!
43//! When a part of the project is finished, copy the files to their "original location". That can be done manually with great control. Do this often and with confidence. The "original" will survive, all the rest is just temporary.
44//!
45//! When the project is finished and all the files are copied into "original location" you do not needed the "temporary working files" any more. Delete these files from the "temporary working copy" to avoid any confusion later. The location of the "originals files" must be always in the same place: the external SSD.
46//!
47//! ## Backup
48//!
49//! In the computer world anything can stop working in a second without any warning signs. I witnessed people loosing all their files because a hard disk died. I saw them cry.
50//!
51//! We must always have a backup of the files. It is tedious and it looks superfluous until the disaster strikes.
52//!
53//! I will use `robocopy` to make the backup or mirror from the external SSD to the old external HDD.
54//!
55//! Never change manually the files on the backup HDD. Only use the robocopy command.
56//!
57//! ## GUI for windows
58//!
59//! My friend is not a computer guy, so I decided that a CLI program in a terminal is not for him. He is comfortable to use GUI programs in Windows.
60//!
61//! I will use the crate `iced` to create a simple GUI program for Windows. It is "retained mode GUI".
62//!
63//! First the program will check what disks are connected. The names of the folders are fixed so I can recognize them easily. The letters of the external disks can get mounted differently, therefore there is an exploration phase.
64//!
65//! 1. x:\original_1
66//! 2. y:\backup_of_original_1

138fn main() {
139    // scaffolding for catch panic and log to file
140    tracing_init();
141
142    let version: &'static str = env!("CARGO_PKG_VERSION");
143    tracing::info!("Start app backup_for_zeljko v{}", version);
144
145    // catch panics and write to log.txt
146    std::panic::set_hook(Box::new(|info| {
147        let backtrace = std::backtrace::Backtrace::force_capture();
148        handle_panic(info.payload(), backtrace)
149    }));
150    let _ = std::panic::catch_unwind(|| {
151        let _ = main_inner();
152    });
153}
154
155fn handle_panic(payload: &(dyn std::any::Any + Send), backtrace: std::backtrace::Backtrace) {
156    tracing::error!("Panicked: ");
157    if let Some(string) = payload.downcast_ref::<String>() {
158        tracing::error!("{string}");
159    } else if let Some(str) = payload.downcast_ref::<&'static str>() {
160        tracing::error!("{str}")
161    } else {
162        tracing::error!("{payload:?}")
163    }
164
165    tracing::error!("Backtrace: {backtrace:#?}");
166}

168/// Initialize tracing to file logs/log.YYYY-MM-DD
169///
170/// The folder logs/ is in .gitignore and will not be committed.
171pub fn tracing_init() {
172    // tracing to file
173    let file_appender = tracing_appender::rolling::daily("logs", "log");
174
175    let offset = time::UtcOffset::current_local_offset().expect("should get local offset!");
176    let timer = tracing_subscriber::fmt::time::OffsetTime::new(
177        offset,
178        time::macros::format_description!("[hour]:[minute]:[second].[subsecond digits:6]"),
179    );
180
181    // Filter out logs from dependency crates
182    // A filter consists of one or more comma-separated directives
183    // target[span{field=value}]=level
184    // examples: wgpu_core=error
185    // directives can be added with the RUST_LOG environment variable:
186    // export RUST_LOG=backup_for_zeljko=trace
187    // Unset the environment variable RUST_LOG
188    // unset RUST_LOG
189    let filter = tracing_subscriber::EnvFilter::from_default_env()
190        .add_directive("backup_for_zeljko=debug".parse().unwrap_or_else(|e| panic!("{e}")))
191        .add_directive("wgpu_core=error".parse().unwrap_or_else(|e| panic!("{e}")))
192        .add_directive("iced_wgpu=error".parse().unwrap_or_else(|e| panic!("{e}")))
193        .add_directive("iced_winit=error".parse().unwrap_or_else(|e| panic!("{e}")))
194        .add_directive("wgpu_hal=error".parse().unwrap_or_else(|e| panic!("{e}")));
195
196    tracing_subscriber::fmt()
197        .with_file(true)
198        .with_max_level(tracing::Level::DEBUG)
199        .with_timer(timer)
200        .with_line_number(true)
201        .with_ansi(false)
202        .with_writer(file_appender)
203        .with_env_filter(filter)
204        .init();
205}

241impl Default for MyApp {
242    /// Defaults are better to show intent then the "constructor" new() method.
243    fn default() -> Self {
244        /// Internal function, because it is called only once and only here.
245        fn find_exist_folder_in_drives(path_wo_drive_letter: &str) -> Option<String> {
246            let drives_letters = vec![
247                "c", "d", "e", "f", "g", "h", "i", "j", "k", "l", "m", "n", "o", "p", "q", "r", "s", "t", "u", "v", "w", "x", "y", "z",
248            ];
249            for x in drives_letters.iter() {
250                let path = format!(r#"{x}:\{path_wo_drive_letter}"#);
251                if std::fs::exists(&path).expect("fs::exists") {
252                    return Some(path);
253                }
254            }
255            None
256        }
257
258        let mut view_page = ViewPage::Main;
259        // External drives can have different letters d:, e:, f:,...
260        // I need to check where is the foldername I expect. The folder names are fixed.
261        let original_1 = find_exist_folder_in_drives(ORIGINAL_1);
262        if original_1.is_none() {
263            view_page = ViewPage::Original1NotFound;
264        }
265        let backup_of_original_1 = find_exist_folder_in_drives(BACKUP_OF_ORIGINAL_1);
266        if backup_of_original_1.is_none() {
267            view_page = ViewPage::BackupOfOriginal1NotFound;
268        }
269        Self {
270            original_1,
271            backup_of_original_1,
272            view_page,
273        }
274    }
275}

448/// This struct is needed just to encapsulate robocopy methods.
449/// Often it is easier to structure the code using struct and impl than using modules.
450/// robocopy d:\original_1 d:\backup_of_original_1 /MIR /FFT
451struct Robocopy {}
452impl Robocopy {
453    /// robocopy MIR
454    pub fn command_robocopy_mir(source: &str, destination: &str) {
455        // /FFT Assumes FAT file times (two-second precision)
456        // /NDL Specifies that directory names are not to be logged.
457        let robocopy_cmd = format!("robocopy {source} {destination} /MIR /FFT /NDL");
458        tracing::info!("{robocopy_cmd}");
459        use std::os::windows::process::CommandExt;
460        std::process::Command::new("cmd.exe")
461            // CREATE_NEW_CONSOLE 0x00000010
462            .creation_flags(0x00000010)
463            // /K - Carries out the command specified by string but remains
464            .arg("/K")
465            .arg(&robocopy_cmd)
466            .spawn()
467            .expect("Command::new")
468            .wait()
469            .unwrap();
470    }
